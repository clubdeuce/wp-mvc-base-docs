413:          * WP save_post action hook callback
414:          *
415:          * @param string $post_id
416:          * @internal
417:          * @access public
418:          * @since 0.1
419:          */
420:         public function callback_save_post( $post_id )
421:         {
422:             $post_type = get_post_type( $post_id );
423:             
424:             // verify if this is an auto save routine. 
425:             // If it is our form has not been submitted, so we dont want to do anything
426:             if ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE )
427:                 return;
428:                 
429:             // We need to check if the current user is authorised to do this action. 
430:             if ( 'page' === $post_type ) {
431:                 if ( ! current_user_can( 'edit_page', $post_id ) )
432:                     return;
433:             } else {
434:                 if ( ! current_user_can( 'edit_post', $post_id ) )
435:                     return;
436:             }
437:             
438:             // Third we need to check if the user intended to change this value.
439:             if ( ! isset( $_POST[ $this->nonce_name ] ) || ! wp_verify_nonce( $_POST[ $this->nonce_name ], $this->nonce_action ) )
440:                 return;
441:             
442:             //If we made it this far, we can save the POST data
443:             //decide the post type, and then call the appropriate save function
444:             switch( $post_type )
445:             {
446:                 case 'post':
447:                     if ( method_exists( $this, 'save_data_post' ) ) {
448:                         return $this->save_data_post( $post_id );
449:                     }
450:                     break;
451:                 case 'page':
452:                     if ( method_exists( $this, 'save_data_page' ) ) {
453:                         return $this->save_data_page( $post_id );
454:                     }
455:                     break;
456:                 default:
457:                     if ( isset( $this->cpts[ $post_type ] ) && method_exists( $this->cpts[ $post_type ], 'save' ) ) {
